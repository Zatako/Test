  <div class="bg"></div>

  <button class="play-btn" id="playBtn" aria-label="播放">
    <svg viewBox="0 0 24 24"><path d="M8 5v14l11-7z"/></svg>
  </button>

  <div class="video-wrap" id="videoWrap">
    <button class="close-btn" id="closeBtn">✕</button>
    <video id="vid" playsinline controls>
      <source src="video.mp4" type="video/mp4"> <!-- 👈 改成你的影片 -->
    </video>
  </div>

<audio id="aud" src="audio.mp3" loop></audio> <!-- 👈 改成你的音訊，不要循環就刪掉 loop -->

  <script>
    const playBtn   = document.getElementById('playBtn');
    const videoWrap = document.getElementById('videoWrap');
    const closeBtn  = document.getElementById('closeBtn');
    const vid       = document.getElementById('vid');
    const aud       = document.getElementById('aud');

    // 點播放鍵 → 播影片
    playBtn.addEventListener('click', () => {
      videoWrap.classList.add('show');
      vid.play().catch(() => {});
    });

    // 影片播完 → 關閉影片，播音訊
    vid.addEventListener('ended', () => {
      videoWrap.classList.remove('show');
      aud.play().catch(() => {});
    });

    // 按關閉 → 關閉影片，播音訊
    closeBtn.addEventListener('click', () => {
      vid.pause();
      vid.currentTime = 0;
      videoWrap.classList.remove('show');
      aud.play().catch(() => {});
    });
  </script>
