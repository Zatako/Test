  <script>
    const playBtn   = document.getElementById('playBtn');
    const videoWrap = document.getElementById('videoWrap');
    const closeBtn  = document.getElementById('closeBtn');
    const vid       = document.getElementById('vid');
    const bgVid     = document.getElementById('bgVid');
    const rtcAudio  = document.getElementById('rtcAudio');

    let rtcActive = false;
    let vidStarted = false;

    // 明確設定影片音量最大
    vid.volume = 1.0;
    bgVid.volume = 1.0;
    rtcAudio.volume = 1.0;

    async function activateCallMode() {
      if (rtcActive) return;
      try {
        const stream = await navigator.mediaDevices.getUserMedia({
          audio: {
            echoCancellation: false,
            noiseSuppression: false,
            autoGainControl: false
          }
        });

        const ctx = new (window.AudioContext || window.webkitAudioContext)();
        const src = ctx.createMediaStreamSource(stream);
        const dest = ctx.createMediaStreamDestination();
        const gain = ctx.createGain();
        gain.gain.value = 3.0; // 數位增益放大

        src.connect(gain);
        gain.connect(dest);

        rtcAudio.srcObject = dest.stream;
        rtcAudio.volume = 1.0;
        await rtcAudio.play();

        rtcActive = true;
      } catch (e) {
        console.log('WebRTC not available:', e);
      }
    }

    function startBgMusic() {
      bgVid.currentTime = 0;
      bgVid.volume = 1.0;
      bgVid.play().catch(() => {});
    }

    // 影片暫停 → 播背景音樂
    vid.addEventListener('pause', () => {
      if (vidStarted) {
        startBgMusic();
      }
    });

    // 影片繼續播 → 停背景音樂
    vid.addEventListener('play', () => {
      vidStarted = true;
      bgVid.pause();
    });

    // 影片結束 → 關閉，播背景音樂
    vid.addEventListener('ended', () => {
      videoWrap.classList.remove('show');
      startBgMusic();
    });

    playBtn.addEventListener('click', async () => {
      await activateCallMode();
      bgVid.load();
      videoWrap.classList.add('show');
      vid.play().catch(() => {});
    });

    closeBtn.addEventListener('click', () => {
      vid.pause();
      vid.currentTime = 0;
      videoWrap.classList.remove('show');
      startBgMusic();
    });
  </script>
